from requests import post as re_post
from requests import get as re_get
from pickle import load,dump
from os import path,listdir

学习卡 6

更新Token

def update_token(user_pass):

api = 'https://api.weixin.qq.com/cgi-bin/token'

params = {

'grant_type':'client_credential',

'appid':user_pass['appid'],

'secret':user_pass['secret']

}

rel = re_get(api,params=params)

if 'access_token' in rel.json():

with open('token_recopt.pk','wb') as file:

dump({'access_token':rel.json()['access_token'],'expires_in':time()},file)

return 0,rel.json()['access_token']

return 1,rel.json()

学习卡 7

上传图片

def upload_img(access_token,img_path):

api = 'https://api.weixin.qq.com/cgi-bin/media/uploadimg'

params = {'access_token':access_token}

img_file = {'media':open(img_path,'rb')}

rel = re_post(api,params=params,files=img_file)

return rel
